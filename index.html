1 Открой QGIS
2 Загрузи слои deforestation.shp LC08_L1TP_182015_20180710_20200831_02_T1_B4_clip.tif LC08_L1TP_182015_20180710_20200831_02_T1_B5_clip.tif LC08_L1TP_182015_20250713_20250713_02_RT_B4_clip.tif LC08_L1TP_182015_20250713_20250713_02_RT_B5_clip.tif
3 Перейди в меню Raster -> Raster Calculator
4 В поле Raster calculator expression введи (LC08_L1TP_182015_20180710_20200831_02_T1_B5_clip@1 - LC08_L1TP_182015_20180710_20200831_02_T1_B4_clip@1) / (LC08_L1TP_182015_20180710_20200831_02_T1_B5_clip@1 + LC08_L1TP_182015_20180710_20200831_02_T1_B4_clip@1)
5 Нажми OK
6 Повтори шаги 3-5 для LC08_L1TP_182015_20250713_20250713_02_RT_B4_clip.tif и LC08_L1TP_182015_20250713_20250713_02_RT_B5_clip.tif
7 Добавь оба полученных NDVI растра в проект
8 Перейди в меню Processing -> Toolbox
9 Найди и запусти функцию Zonal Statistics
10 Выбери первый NDVI растр, затем слой deforestation.shp и укажи имя поля для сохранения статистики
11 Повтори шаги 9-10 для второго NDVI растра

Решение на Python:

import os
from osgeo import gdal, ogr

def calculate_ndvi(red_band_path, nir_band_path):
    red = gdal.Open(red_band_path).ReadAsArray().astype(float)
    nir = gdal.Open(nir_band_path).ReadAsArray().astype(float)
    ndvi = (nir - red) / (nir + red)
    return ndvi

def get_zonal_stats(vector_path, raster_data):
    driver = ogr.GetDriverByName('ESRI Shapefile')
    dataSource = driver.Open(vector_path, 1)
    layer = dataSource.GetLayer()
    for feature in layer:
        geom = feature.GetGeometryRef()
        minx, maxx, miny, maxy = geom.GetEnvelope()
        x_res = (maxx - minx) / raster_data.shape[1]
        y_res = (maxy - miny) / raster_data.shape[0]
        transform = (minx, x_res, 0, maxy, 0, -y_res)
        mem_driver = gdal.GetDriverByName('MEM')
        mem_raster = mem_driver.Create('', raster_data.shape[1], raster_data.shape[0], 1, gdal.GDT_Float32)
        mem_raster.SetGeoTransform(transform)
        mem_raster.GetRasterBand(1).WriteArray(raster_data)
        cutlineDSName = '/vsimem/polygon.shp'
        gdal.VectorTranslate(cutlineDSName, vector_path, format='Memory', layerCreationOptions=['GEOMETRY_NAME=geom'])
        gdal.Warp('/vsimem/clipped.tif', mem_raster, cutlineDS=cutlineDSName, cropToCutline=True)
        clipped_raster = gdal.Open('/vsimem/clipped.tif').ReadAsArray()
        mean_value = clipped_raster[clipped_raster != 0].mean()
        feature.SetField('ndvi', mean_value)
        layer.SetFeature(feature)
    dataSource.Destroy()

vector_path = 'deforestation.shp'
red_band_2018 = 'LC08_L1TP_182015_20180710_20200831_02_T1_B4_clip.tif'
nir_band_2018 = 'LC08_L1TP_182015_20180710_20200831_02_T1_B5_clip.tif'
red_band_2025 = 'LC08_L1TP_182015_20250713_20250713_02_RT_B4_clip.tif'
nir_band_2025 = 'LC08_L1TP_182015_20250713_20250713_02_RT_B5_clip.tif'

ndvi_2018 = calculate_ndvi(red_band_2018, nir_band_2018)
ndvi_2025 = calculate_ndvi(red_band_2025, nir_band_2025)

get_zonal_stats(vector_path, ndvi_2018)
get_zonal_stats(vector_path, ndvi_2025)

driver = ogr.GetDriverByName('ESRI Shapefile')
dataSource = driver.Open(vector_path, 1)
layer = dataSource.GetLayer()
layer_def = layer.GetLayerDefn()
field_index = layer_def.GetFieldIndex('ndvi')

increased_count = 0
for feature in layer:
    ndvi_2018 = feature.GetField(field_index)
    feature.SetField('ndvi', 0)
    layer.SetFeature(feature)
    ndvi_2025 = feature.GetField(field_index)
    if ndvi_2025 > ndvi_2018:
        increased_count += 1

print(increased_count)
